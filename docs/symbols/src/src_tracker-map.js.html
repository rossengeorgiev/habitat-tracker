  7 /**
  8  * Creates an instance of a Map.
  9  *
 10  * @this {habitat.tracker.Map}
 11  * @extends {google.maps.Map} google.maps.Map
 12  * @param {HTMLElement} element DOM element that should be used for the map.
 13  * @param {Object.<string, *>=} opt_options MapOptions.
 14  * @see <a href="https://developers.google.com/maps/documentation/javascript
 15  * /reference#MapOptions">google.maps.MapOptions</a>
 16  * @constructor
 17  * @export
 18  */
 19 habitat.tracker.Map = function(element, opt_options) {
 20   var options = {
 21     center: new google.maps.LatLng(52, 0.5),
 22     zoom: 8,
 23     mapTypeId: google.maps.MapTypeId.ROADMAP
 24   };
 25   goog.object.extend(options, opt_options || {});
 26 
 27   google.maps.Map.call(this, element, options);
 28 };
 29 goog.inherits(habitat.tracker.Map, google.maps.Map);

 32 /**
 33  * Converts from LatLng to pixel position.
 34  *
 35  * @export
 36  * @this {habitat.tracker.Map}
 37  * @param {google.maps.LatLng} position Position to convert.
 38  * @return {google.maps.Point} Pixel position.
 39  */
 40 habitat.tracker.Map.prototype.fromLatLngToPixel = function(position) {
 41   var scale = Math.pow(2, this.getZoom());
 42   var proj = this.getProjection();
 43   var bounds = this.getBounds();
 44 
 45   var nw = proj.fromLatLngToPoint(
 46       new google.maps.LatLng(
 47           bounds.getNorthEast().lat(),
 48           bounds.getSouthWest().lng()
 49       ));
 50   var point = proj.fromLatLngToPoint(position);
 51 
 52   return new google.maps.Point(
 53       Math.floor((point.x - nw.x) * scale),
 54       Math.floor((point.y - nw.y) * scale));
 55 };

 58 /**
 59  * Converts from pixel position to LatLng.
 60  *
 61  * @export
 62  * @this {habitat.tracker.Map}
 63  * @param {google.maps.Point} pixel Position to convert.
 64  * @return {google.maps.LatLng} LatLng position.
 65  */
 66 habitat.tracker.Map.prototype.fromPixelToLatLng = function(pixel) {
 67   var scale = Math.pow(2, this.getZoom());
 68   var proj = this.getProjection();
 69   var bounds = this.getBounds();
 70 
 71   var nw = proj.fromLatLngToPoint(
 72       new google.maps.LatLng(
 73           bounds.getNorthEast().lat(),
 74           bounds.getSouthWest().lng()
 75       ));
 76   var point = new google.maps.Point(
 77       pixel.x / scale + nw.x,
 78       pixel.y / scale + nw.y);
 79 
 80   return proj.fromPointToLatLng(point);
 81 };

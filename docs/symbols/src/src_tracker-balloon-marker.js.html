  7 /**
  8  * Creates a new BalloonMarker.
  9  *
 10  * @this {habitat.tracker.BalloonMarker}
 11  * @extends {google.maps.Marker} google.maps.Marker
 12  * @param {Object.<string, *>=} opt_options MarkerOptions.
 13  * @see <a href="https://developers.google.com/maps/documentation/javascript
 14  * /reference#MarkerOptions">google.maps.MarkerOptions</a>
 15  * @constructor
 16  * @export
 17  */
 18 habitat.tracker.BalloonMarker = function(opt_options) {
 19   var options = {
 20     optimized: false, // make retina images work
 21     altitude: 0,
 22     image: null,
 23     shadow: /** @type {string} */ {
 24       url: 'images/markers/shadow.png',
 25       anchor: new google.maps.Point(
 26           habitat.tracker.BalloonMarker.SHADOW_SIZE_.width / 2,
 27           habitat.tracker.BalloonMarker.SHADOW_SIZE_.height / 2),
 28       scaledSize: habitat.tracker.BalloonMarker.SHADOW_SIZE_
 29     }
 30   };
 31   goog.object.extend(options, opt_options || {});
 32 
 33   google.maps.Marker.call(this, options);
 34 };
 35 goog.inherits(habitat.tracker.BalloonMarker, google.maps.Marker);

 89 /**
 90  * Called by MVCObject when image is changed.
 91  *
 92  * @protected
 93  * @export
 94  */
 95 habitat.tracker.BalloonMarker.prototype.image_changed = function() {
 96   var image_ = /** @type {string} */ this.get('image');
 97   var this_ = this;
 98   this.getImageSize_(image_, function(width, height) {
 99     this_.image_size_ = new google.maps.Size(width / 2, height / 2); // retina
100     this_.setIcon(/** @type {google.maps.Icon} */ {
101       url: image_,
102       anchor: new google.maps.Point(
103           this_.image_size_.width / 2,
104           this_.image_size_.height + this_.getPixelAltitude_()),
105       scaledSize: this_.image_size_
106     });
107   });
108 };

111 /**
112  * Called by MVCObject when image is changed.
113  *
114  * @protected
115  * @export
116  */
117 habitat.tracker.BalloonMarker.prototype.altitude_changed = function() {
118   if (this.image_size_ == undefined || this.image_size_ == null)
119     return;
120   var altitude = /** @type {number} */ this.get('altitude');
121   var icon = /** @type {google.maps.Icon} */ this.getIcon();
122   icon.anchor = new google.maps.Point(
123       this.image_size_.width / 2,
124       this.image_size_.height + this.getPixelAltitude_());
125 };

128 /**
129  * Retrieves image size.
130  *
131  * @param {string} src Image URL.
132  * @param {function(number, number)} func
133  *        Function called when image size is retrieved.
134  * @private
135  */
136 habitat.tracker.BalloonMarker.prototype.getImageSize_ = function(src, func) {
137   var img = new Image();
138   img.src = src;
139   img.onload = function() {
140     if (func != undefined && func != null)
141       func(img.width, img.height);
142   };
143 };

146 /**
147  * Gets the altitude of the balloon in pixels.
148  *
149  * @return {number} Altitude in pixels.
150  * @private
151  */
152 habitat.tracker.BalloonMarker.prototype.getPixelAltitude_ = function() {
153   var altitude = /** @type {number} */ this.get('altitude');
154   return Math.round(altitude / (1000 / 3) * (this.getMap().getZoom() / 18.0));
155 };
